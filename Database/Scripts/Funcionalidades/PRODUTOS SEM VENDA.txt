SELECT A.CODIGO, A.NOME, B.EST_ATUAL, B.DAT_COMPRA, B.DAT_VENDA FROM PRODUTOS A
INNER JOIN PRODLOJAS B ON A.CODIGO = B.CODIGO
WHERE A.ATIVO <> 'N'
AND B.CODLOJA = 1
AND B.CODIGO NOT IN (
	SELECT CODPRODUTO FROM NVENDI2 A 
	INNER JOIN NVENDA2 B ON A.NUMVENDA = B.CODIGOVENDA 
	WHERE B.EMISSAO BETWEEN '2018-01-01' AND '2021-07-26')
AND B.EST_ATUAL <= 0



select SUM(B.NVTOTAL) from produtos a 
inner join nvendi2 b on a.codigo = b.CODPRODUTO
inner join nvenda2 c on b.numvenda = c.codigovenda
where c.emissao between '2021-07-01' and '2021-07-29'
AND a.GRUPO = 5
AND C.O_V = 2
