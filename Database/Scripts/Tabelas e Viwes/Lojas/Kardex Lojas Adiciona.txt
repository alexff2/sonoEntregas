UNION
SELECT 1 AS LOJA, B.EMISSAO AS DATA, CONVERT(VARCHAR(8), '23:00:00') AS HORA, CONVERT(VARCHAR(30), 'ENTREGA') AS MODULO, 
CONVERT(VARCHAR(15), CONVERT(numeric, A.NUMVENDA)) AS DOCUMENTO, A.CODPRODUTO, A.QUANTIDADE AS QUANT_ENTRADA, 
0 AS QUANT_SAIDA, CONVERT(CHAR(1), 'E') AS ENTRADA_SAIDA, CONVERT(VARCHAR(80), 'Modulo de entregas') AS OBS, 
B.CODCLIENTE AS CODCF, B.NOMECLI AS NOMECF, 1 AS EMBALAGEM, CONVERT(VARCHAR(1), 'S') AS CALCULAR, A.UNITARIO1 AS VALORUNITARIO, 
CONVERT(VARCHAR(30), NULL) AS LOTE, CONVERT(VARCHAR(30), 'Cliente') AS TIPOCLI_FOR_FUN,
CONVERT(VARCHAR(30), 'Padrao') AS USUARIO, 1 AS ITEM, -1 AS MU_ITEM, 1 AS MU_QTE_MENORUN, CONVERT(CHAR(3), 'UN') MU_UNIDADE
FROM NVENDI2 A
INNER JOIN NVENDA2 B ON A.NUMVENDA = B.CODIGOVENDA
WHERE A.STATUS IS NOT NULL
